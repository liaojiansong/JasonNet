产品总数据api

API功能 获取产品的设备,触发器,数据统计数据
请求方法 POST
请求地址 http://jasonnet.com/index/api/product
请求参数

// 请求设备数据统计

{

// 令牌 = api + 密码 => md5()

"api_key": "MTUyNTU3NDI3MzY2MTUw",

// 产品id

"product_id": 1,

// 请求类型

"type": "get_statistical_info"

}
返回示例

// 返回示例

{

"code": 10000,

"error": "success",

"data": {

"product_id": 1,

"product_name": 30,

"device_count": 1,

"trigger_count":60

}

}

新增设备

API功能 为指定产品添加设备
请求方法 POST
请求地址 http://jasonnet.com/index/api/device
请求参数 
                                      // 新增设备
  {
    "api_key": "MTUyNTU3NDI3MzY2MTUw",
    // 产品id
    "product_id": 1,
    // 请求类型
    "type": "add_device",
    // 设备信息
    "data": {
      // 设备名称
      "device_name": "小米8",
      // 鉴权信息
      "device_auth": "mi_8",
      // 设备描述（可选）
      "device_description": "奋斗联盟的一员"
    }
  },
返回示例 
  {
    "errno": 10000,
    "error": "success",
    "data": {
      //平台分配的设备ID
      "device_id": "156888"
    }
  },

更新设备

API功能 新增触发器
请求方法 POST
请求地址 http://jasonnet.com/index/api/trigger
请求参数 
  {
    "api_key": "MTUyNTU3NDI3MzY2MTUw",
    // 产品id
    "product_id": 2,
    // 请求类型
    "type": "add_trigger",
    // 触发器信息
    "data": {
      // 设备id
      "device_id": 8,
      // 触发器名称
      "trigger_name": "高温报警",
      // 判断条件
      "target_condition": ">",
      // >= | < | <= | == | change,
      // 阈值
      "target_value": 60,
      // 目标邮箱
      "email": "729999@gamil.com",
      // 报警信息（可选）
      "report_msg": "设备温度过高,请检查"
    }
  }
返回示例 
  {
    "code": 10000,
    "error": "success",
    "data": {
      //平台分配的触发器ID
      "trigger_id": "28985"
    }
  }
